56 static degree_abstract azimuth(earth_point from_point, earth_point to_point)
57 {
58  double w1 = from_point.latitude().to_radians();
59  double d1 = from_point.longitude().to_radians();
60  double w2 = to_point.latitude().to_radians();
61  double d2 = to_point.longitude().to_radians();
62 
63  double cosab = sin(w1) * sin(w2) + cos(w1) * cos(w2) * cos(d2 - d1);
64  double x = sin(d2 - d1) * sin(pi / 2. - w1) * sin(pi / 2. - w2);
65  double y = cos(pi / 2. - w2) - cos(pi / 2. - w1) * cosab;
66  double az = atan2(x, y);

68  az += pi * 2;
