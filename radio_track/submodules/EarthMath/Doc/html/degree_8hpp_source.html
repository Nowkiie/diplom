775 inline int64_t latitude_functional::normalize(int64_t value)
776 {
777  int32_t sign;
778  value < 0 ? sign = -1 : sign = 1;
779  value = abs(value);
780  int64_t degr360 = (int64_t(360) * minutes_in_degrees * seconds_in_minute
781  * milli_in_second);
782  int64_t degr180 = degr360 / 2;
783  int64_t degr90 = degr360 / 4;
784  value %= degr360;
785  if (value > degr180)
786  {
787  sign *= -1;
788  value -= degr180;
789  }
790  if (value > degr90)
791  value = degr180 - value;
792  return sign * value;
793 }
